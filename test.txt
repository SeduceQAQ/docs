services:

    depends_on:
      halodb:
        condition: service_healthy
    networks:
      halo_network:
    volumes:

    ports:

    healthcheck:

      timeout: 5s
      retries: 5

    command:

      - --spring.r2dbc.username=halo
      # PostgreSQL 的密码，请保证与下方 POSTGRES_PASSWORD 的变量值一致。

      - --halo.external-url=https://blog.seduceqaq.com:8090/

      - --halo.security.initializer.superadminusername=seduce
      # 初始化的超级管理员密码
      - --halo.security.initializer.superadminpassword=@Duolaameng6
  halodb:
    image: postgres:latest
    container_name: halodb
    restart: on-failure:3
    networks:
      halo_network:
    volumes:
      - ./db:/root/postgresql/data
    ports:
      - "5432"
    healthcheck:
      test: [ "CMD", "pg_isready" ]
      interval: 10s
      timeout: 5s
      retries: 5
    environment:
      - POSTGRES_PASSWORD=@Duolaameng6
      - POSTGRES_USER=halo
      - POSTGRES_DB=halo
      - PGUSER=halo
